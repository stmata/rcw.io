async def get_temperature(ville: str, api_key: str) -> TemperatureResponse:
    """
    Interroge OpenWeatherMap et retourne la temp√©rature
    d'une ville donn√©e.
    
    Args:
        ville: Nom de la ville √† rechercher
        api_key: Cl√© API OpenWeatherMap
    
    Returns:
        TemperatureResponse avec les donn√©es m√©t√©o
    
    Raises:
        HTTPException: Si la ville n'est pas trouv√©e ou
                       si l'API externe √©choue
    """
    
    # Param√®tres de la requ√™te vers l'API externe

        "q": ville,

    }

    # ‚îÄ‚îÄ Appel asynchrone √† l'API externe ‚îÄ‚îÄ

                detail="L'API m√©t√©o n'a pas r√©pondu √† temps."

        except httpx.RequestError as e:

    # ‚îÄ‚îÄ Gestion des codes de statut HTTP ‚îÄ‚îÄ

            status_code=404,

            status_code=500,
            detail="Cl√© API invalide. V√©rifiez votre configuration."
        )

            status_code=502,

        )

    # ‚îÄ‚îÄ Extraction et transformation des donn√©es ‚îÄ‚îÄ

        temperature=round(data["main"]["temp"], 1),

async def fetch_json_as_dataframe(url: str) -> pd.DataFrame:
    """
    R√©cup√®re des donn√©es JSON depuis une URL et les
    convertit en DataFrame Pandas.

    Args:
        url: URL retournant un tableau JSON (list of objects)

    """

            detail="Le JSON doit √™tre un tableau (liste d'objets)."
        )
